examples  = transpose(r_examples)
allTools  = r_all_tools
allKws    = r_all_keywords
submitUrl = r_submit_url
html`
<div class="dash-header">
  <div class="dash-header-inner">
    <div class="dash-eyebrow">Resource Library</div>
    <h1 class="dash-title">Examples Dashboard</h1>
    <p class="dash-subtitle">
      Discover community-contributed examples across tools and topics.
      Filter below, or
      <a href="${submitUrl}" target="_blank" rel="noopener" class="submit-link">
        submit your own ↗
      </a>
    </p>
  </div>
</div>`
viewof rawSearch = Inputs.text({
  placeholder: "Search by title, description, author…",
  label: null,
  submit: false,
  attributes: { "aria-label": "Search examples" }
})

// ── Tools multi-select ────────────────────────────────────────────────────────
viewof rawTools = Inputs.select(allTools, {
  label: "Filter by Tool",
  multiple: true,
  size: Math.min(allTools.length, 8),
  attributes: { "aria-label": "Filter by tool type" }
})

// ── Keywords multi-select ─────────────────────────────────────────────────────
viewof rawKeywords = Inputs.select(allKws, {
  label: "Filter by Keyword",
  multiple: true,
  size: Math.min(allKws.length, 8),
  attributes: { "aria-label": "Filter by keyword" }
})
selectedTools    = Array.from(rawTools)
selectedKeywords = Array.from(rawKeywords)
searchQuery      = rawSearch.toLowerCase().trim()

// Filter examples — OR logic within tools, OR logic within keywords
filtered = examples.filter(ex => {
  // 1. Text search across title, description, author, tags
  const textMatch =
    !searchQuery ||
    ex.title.toLowerCase().includes(searchQuery) ||
    ex.description.toLowerCase().includes(searchQuery) ||
    (ex.author || "").toLowerCase().includes(searchQuery) ||
    ex.tools.toLowerCase().includes(searchQuery) ||
    ex.keywords.toLowerCase().includes(searchQuery)

  // 2. Tools filter: passes if no tools selected, OR example has ≥1 match
  const exTools  = ex.tools ? ex.tools.split("|").map(t => t.trim()) : []
  const toolMatch =
    selectedTools.length === 0 ||
    selectedTools.some(t => exTools.includes(t))

  // 3. Keywords filter: same OR logic
  const exKws   = ex.keywords ? ex.keywords.split("|").map(k => k.trim()) : []
  const kwMatch =
    selectedKeywords.length === 0 ||
    selectedKeywords.some(k => exKws.includes(k))

  return textMatch && toolMatch && kwMatch
})
function clearText(viewofEl) {
  const input = viewofEl.querySelector("input")
  if (input) { input.value = ""; input.dispatchEvent(new Event("input", { bubbles: true })) }
}

function clearSelect(viewofEl) {
  const select = viewofEl.querySelector("select")
  if (select) {
    Array.from(select.options).forEach(o => { o.selected = false })
    select.dispatchEvent(new Event("input", { bubbles: true }))
  }
}
html`
<div class="results-bar">
  <span class="results-count">
    ${filtered.length === examples.length
      ? `Showing all <strong>${examples.length}</strong> examples`
      : `Showing <strong>${filtered.length}</strong> of ${examples.length} examples`}
  </span>
  ${(selectedTools.length > 0 || selectedKeywords.length > 0 || searchQuery)
    ? html`<button class="clear-btn" onclick=${() => {
        clearText(viewof rawSearch)
        clearSelect(viewof rawTools)
        clearSelect(viewof rawKeywords)
      }}>✕ Clear all filters</button>`
    : ""}
</div>`
function renderTag(label, type) {
  return html`<span class="tag tag-${type}">${label}</span>`
}

// ── Card renderer ──────────────────────────────────────────────────────────────
function renderCard(ex) {
  const toolTags = ex.tools
    ? ex.tools.split("|").map(t => renderTag(t.trim(), "tool"))
    : []
  const kwTags = ex.keywords
    ? ex.keywords.split("|").map(k => renderTag(k.trim(), "keyword"))
    : []

  const hasImage = ex.image &&
    !ex.image.endsWith("/") &&
    ex.image !== ""

  return html`
    <article class="ex-card">
      <div class="card-img-wrap ${hasImage ? "" : "card-img-placeholder"}">
        ${hasImage
          ? html`<img
               src="${ex.image}"
               alt="${ex.title}"
               class="card-img"
               loading="lazy"
               onerror="this.parentElement.classList.add('card-img-placeholder'); this.remove()"
             />`
          : html`<span class="placeholder-icon" aria-hidden="true">◈</span>`}
      </div>
      <div class="card-body">
        <h3 class="card-title">
          <a href="${ex.url}" target="_blank" rel="noopener">${ex.title}</a>
        </h3>
        <p class="card-desc">${ex.description}</p>
        ${ex.author
          ? html`<p class="card-author">by ${ex.author}</p>`
          : ""}
        <div class="card-tags">
          ${toolTags}
          ${kwTags}
        </div>
      </div>
    </article>`
}

// ── Render grid or empty state ─────────────────────────────────────────────────
filtered.length === 0
  ? html`
    <div class="empty-state">
      <div class="empty-icon" aria-hidden="true">◈</div>
      <p>No examples match your current filters.</p>
      <p>
        <a href="${submitUrl}" target="_blank" rel="noopener">
          Know one that should be here? Submit it ↗
        </a>
      </p>
    </div>`
  : html`<div class="card-grid">${filtered.map(renderCard)}</div>`
html`
<div class="submit-cta">
  <div class="submit-cta-inner">
    <div class="submit-icon" aria-hidden="true">✦</div>
    <h2>Know a great example?</h2>
    <p>
      Contributions from the community keep this dashboard growing.
      Submit a resource using our GitHub Issue template — an admin will
      review it and add it to the next site build.
    </p>
    <a href="${submitUrl}" target="_blank" rel="noopener" class="submit-btn">
      Submit an Example ↗
    </a>
  </div>
</div>`
